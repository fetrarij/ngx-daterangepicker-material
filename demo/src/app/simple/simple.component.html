Simple daterangepicker

Using the change event : 
<input type="text" ngxDaterangepickerMd
[(ngModel)]="selected" placeholder="Choose date" (change)="change($event)"/>
Or with ngModel : 
<input type="text" ngxDaterangepickerMd
        [lockStartDate]="false"
[(ngModel)]="selected" placeholder="Choose date"
(ngModelChange)="ngModelChange($event)"/>
